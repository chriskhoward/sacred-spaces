
import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: ['/studio/', '/api/', '/dashboard/', '/onboarding/', '/user-profile/'],
    },
    sitemap: 'https://flowinfaith.com/sitemap.xml',
  }
}
